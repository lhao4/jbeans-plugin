import com.sun.tools.attach.AttachNotSupportedException;
import com.sun.tools.attach.VirtualMachine;
import com.sun.tools.attach.VirtualMachineDescriptor;
import java.util.List;

public class JvmProcessScanner {

    public List<VirtualMachineDescriptor> scanAllJavaProcesses() {
        return VirtualMachine.list();
    }

    // 判断是否是 SpringBoot 应用
    public boolean isSpringBootProcess(VirtualMachineDescriptor vmd) {
        String displayName = vmd.displayName();
        return displayName.contains("spring-boot") || displayName.contains(".jar");
    }
}

## 模块 4：Spring 上下文获取（通用无侵入）
```java
public class SpringContextFinder {
    public static ApplicationContext find() {
        try {
            // 从静态变量获取Spring上下文（通用）
            Field field = ContextLoader.class.getDeclaredField("currentContext");
            field.setAccessible(true);
            return (ApplicationContext) field.get(null);
        } catch (Exception e) {
            // 兼容SpringBoot
            return SpringBootContextFinder.find();
        }
    }
}
```

## 模块 5：万能方法执行器（核心！支持所有方法）
### 支持
- 静态方法
- Spring Bean 方法
- 普通类对象方法
- Dubbo 接口
- HTTP 接口

```java
public class UniversalMethodExecutor {

    /**
     * 统一执行任意方法
     */
    public ExecuteResult invoke(InvokeRequest request) {
        try {
            if (request.isStatic()) {
                // 静态方法直接执行
                return invokeStatic(request);
            }

            // 尝试从Spring获取Bean
            Object bean = SpringBeanFinder.getBean(request.getClassName());
            if (bean != null) {
                return invokeSpringBean(bean, request);
            }

            // 普通类 -> 创建对象执行
            Object instance = Class.forName(request.getClassName()).newInstance();
            return invokeInstanceMethod(instance, request);
        } catch (Exception e) {
            return ExecuteResult.fail(e.getMessage());
        }
    }
}
```
